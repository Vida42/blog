```python
class Solution:
    def decodeString(self, s: str) -> str:
        curNum = 0
        curStr = ""
        stack = []
        for each in s:
            # if is digit, formatting whole number
            if each.isdigit():
                curNum = 10 * curNum + int(each)
            # time to decode
            elif each == ']':
                curNum = stack.pop()
                prevStr = stack.pop()
                curStr = prevStr + curStr * curNum
                curNum = 0
            # to receive new nested string
            elif each == '[':
                stack.append(curStr)
                stack.append(curNum)
                curStr = ""
                curNum = 0
            # string
            else:
                curStr += each
        return curStr
```
