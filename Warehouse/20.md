```python
# 52 ms, faster than 99.27%

class Solution(object):
	def isValid(self, s):
		"""
		:type s: str
		:rtype: bool
		"""
		result = [' ']
		dic = {')':'(',']':'[','}':'{'}
		for i in s:
			if i in dic and dic[i] == result[-1]:
				result.pop()
			else:
				result.append(i)
		return len(result) == 1
```

```python
class Solution:
    def isValid(self, s: str) -> bool:
        lookup = {'(':')', '[':']', '{':'}'}
        stack = []
        for each in s:
            if each in lookup:
                stack.append(each)
            else:
                if stack and lookup[stack[-1]] == each:
                    stack.pop()
                else:
                    return False
        if stack:
            return False
        return True
```
